## 1. Architecture Comparison

### OpenCelerator (Reference) vs SCC (Current)

| Component | OpenCelerator (Python/Qt) | SCC (JavaScript/Flask) |
|-----------|---------------------------|------------------------|
| **Backend** | Python + pandas DataFrames | Flask (minimal) + client-side JS |
| **Data Storage** | SQLite database | IndexedDB (browser) |
| **UI Framework** | PyQt5 dialogs | Tailwind CSS + custom toaster |
| **Event System** | EventStateManager pub/sub | eventBus.js pub/sub |
| **File Parsing** | pandas (CSV, Excel, ODS) | None currently (need xlsx, papaparse) |
| **Data Model** | DataFrame (df_raw) | chartState.series arrays |

### Key Architectural Difference

**OpenCelerator**: Server-side data processing with DataFrames
```
File → pandas.read_csv() → DataFrame → column mapping → cleaning → df_raw → database
```

**SCC**: Client-side processing with arrays
```
File → FileReader API → xlsx/papaparse → arrays → column mapping → cleaning → chartState.series → IndexedDB
```

### Mapping to OpenCelerator Column System

| OpenCelerator | SCC Equivalent | Notes |
|---------------|----------------|-------|
| `'d'` (date) | `xValues` | Unix timestamps, converted via dates.js |
| `'c'` (correct) | `corrects` | Positive metric |
| `'i'` (incorrect) | `errors` | Negative metric |
| `'m'` (minutes) | `timing` | Duration for frequency calculation |
| `'o0', 'o1'...` | `misc.misc1`, `misc.misc2`... | Dynamic columns (max 10) |

### Key Difference: No Column Map Needed

In SCC, the internal column structure is **fixed** (`xValues`, `corrects`, `errors`, `timing`, `misc`). The column mapping dialog needs to map **user's spreadsheet columns** → **these fixed system columns**.

### Phase 2: File Reading

```javascript
// static/SCC/import/fileReader.js

async function readSpreadsheetFile(file) {
    const ext = file.name.split('.').pop().toLowerCase();

    if (['xlsx', 'xls', 'ods'].includes(ext)) {
        return readExcelFile(file);
    } else {
        return readCSVFile(file);
    }
}

async function readExcelFile(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = (e) => {
            try {
                const data = new Uint8Array(e.target.result);
                const workbook = XLSX.read(data, { type: 'array' });
                const firstSheet = workbook.Sheets[workbook.SheetNames[0]];
                resolve(XLSX.utils.sheet_to_json(firstSheet));
            } catch (err) {
                // Fallback to CSV
                readCSVFile(file).then(resolve).catch(reject);
            }
        };
        reader.readAsArrayBuffer(file);
    });
}

async function readCSVFile(file) {
    return new Promise((resolve, reject) => {
        Papa.parse(file, {
            header: true,
            skipEmptyLines: true,
            complete: (results) => resolve(results.data),
            error: reject
        });
    });
}
```

### Phase 3: Column Detection

```javascript
// static/SCC/import/columnDetection.js

const DATE_PATTERN = /^(?=.*\d{2})(?:[^-/.\n]*[-/.]){2,}[^-/.\n]*$/;
const NUMERIC_PATTERN = /^\s*-?\d+(\.\d+)?\s*$/;

function lazyCheck(data, pattern, threshold = 0.8, checkLimit = 10) {
    const columns = Object.keys(data[0] || {});
    const matchingColumns = [];

    for (const col of columns) {
        const toCheck = data.slice(0, checkLimit)
            .map(row => row[col])
            .filter(v => v != null && v !== '');

        if (toCheck.length === 0) continue;

        const matches = toCheck.filter(v => pattern.test(String(v))).length;
        if (matches / toCheck.length >= threshold) {
            matchingColumns.push(col);
        }
    }

    return matchingColumns;
}

function detectColumnTypes(data) {
    const dateColumns = lazyCheck(data, DATE_PATTERN);
    const numericColumns = lazyCheck(data, NUMERIC_PATTERN)
        .filter(col => !dateColumns.includes(col));

    return { dateColumns, numericColumns };
}
```

### Phase 5: Data Cleaning Pipeline

```javascript
// static/SCC/import/dataCleaning.js

export function cleanImportedData(rawData, columnMap) {
    // Step 1: Rename columns
    const renamedData = rawData.map(row => ({
        d: row[columnMap.d],
        c: row[columnMap.c],
        i: row[columnMap.i],
        m: columnMap.m ? row[columnMap.m] : 1  // Default timing = 1
    }));

    // Step 2: Parse dates
    for (const row of renamedData) {
        row.d = parseDate(row.d);
    }

    // Step 3: Clean numeric columns
    for (const row of renamedData) {
        row.c = cleanNumeric(row.c);
        row.i = cleanNumeric(row.i);
        row.m = cleanNumeric(row.m) || 1;
    }

    // Step 4: Filter invalid rows
    return renamedData.filter(row =>
        row.d instanceof Date && !isNaN(row.d) &&
        (row.c !== null || row.i !== null)
    );
}

function parseDate(value) {
    const str = String(value).trim();

    // Year only: "2024" → last day of year
    if (/^\d{4}$/.test(str)) {
        return new Date(`${str}-12-31`);
    }

    // Year-month: "2024-03" → last day of month
    const ymMatch = str.match(/^(\d{4})[-/.](\d{1,2})$/);
    if (ymMatch) {
        const [, year, month] = ymMatch;
        return new Date(year, parseInt(month), 0);  // Day 0 = last day of prev month
    }

    // Standard date parsing
    return new Date(str);
}

function cleanNumeric(value) {
    if (value == null || value === '') return null;
    const num = parseFloat(String(value).trim());
    return (isNaN(num) || num < 0) ? null : num;
}
```

### Phase 6: Integration with chartState

```javascript
// static/SCC/import/dataImport.js

import { chartState } from '../chartState.js';
import { eventBus, EVENTS } from '../eventBus.js';
import { readSpreadsheetFile } from './fileReader.js';
import { detectColumnTypes } from './columnDetection.js';
import { createColumnMappingDialog } from './columnMappingDialog.js';
import { cleanImportedData } from './dataCleaning.js';

export async function importDataFromFile(file) {
    eventBus.emit(EVENTS.DATA_IMPORT_STARTED);

    try {
        // Step 1: Read file
        const rawData = await readSpreadsheetFile(file);

        // Step 2: Detect column types
        const { dateColumns, numericColumns } = detectColumnTypes(rawData);

        // Step 3: Show column mapping dialog
        const columnMap = await new Promise((resolve, reject) => {
            createColumnMappingDialog({
                columns: Object.keys(rawData[0] || {}),
                dateColumns,
                numericColumns,
                onConfirm: resolve,
                onCancel: () => reject(new Error('Cancelled'))
            });
        });

        // Step 4: Clean data
        const cleanedData = cleanImportedData(rawData, columnMap);

        // Step 5: Insert into chartState
        for (const row of cleanedData) {
            const timestamp = Math.floor(row.d.getTime() / 1000);
            chartState.series.xValues.push(timestamp);
            chartState.series.corrects.push(row.c ?? NaN);
            chartState.series.errors.push(row.i ?? NaN);
            chartState.series.timing.push(row.m);
        }

        // Step 6: Emit completion event
        eventBus.emit(EVENTS.DATA_IMPORT_COMPLETED, {
            rowCount: cleanedData.length
        });
        eventBus.emit(EVENTS.DATA_CHART_REFRESH);

    } catch (error) {
        eventBus.emit(EVENTS.DATA_IMPORT_FAILED, { error: error.message });
    }
}
```
